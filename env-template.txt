# Environment Configuration Template
# Create a .env file in the dist folder with this content:

# Required: Your Gemini API key from Google AI Studio
# Get your key at: https://aistudio.google.com/app/apikey
GEMINI_API_KEY=your_gemini_api_key_here

# Optional: Application settings (defaults are usually fine)
APP_NAME=Windows Service Host
STEALTH_MODE=true

# Instructions:
# 1. Copy this file to 'dist/.env'
# 2. Replace 'your_gemini_api_key_here' with your actual API key
# 3. Save the file
# 4. Run StealthAssistant.exe
